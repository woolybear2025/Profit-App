// ─── Calc helpers ─────────────────────────────────────────────────────────────
function calcIngCost(ing) {
const tg=parseFloat(ing.totalGrams), tp=parseFloat(ing.totalPrice), ug=parseFloat(ing.usedGrams);
if(!tg||!tp||!ug||tg===0) return 0;
return (tp/tg)*ug;
}
function calcProd(p) {
const ingCost = p.ingredients.reduce((s,i)=>s+calcIngCost(i),0);
const pkg=parseFloat(p.packagingCost)||0, labor=parseFloat(p.laborCost)||0;
const totalCost=ingCost+pkg+labor, qty=parseFloat(p.outputQty)||1;
const costPerUnit=qty>0?totalCost/qty:0, sell=parseFloat(p.sellPrice)||0;
const profitPerUnit=sell-costPerUnit, margin=sell>0?(profitPerUnit/sell)*100:0;
return {ingCost,totalCost,costPerUnit,profitPerUnit,margin,qty,sell};
}
function calcBatch(ingredients) {
const valid=ingredients.filter(i=>parseFloat(i.totalGrams)>0&&parseFloat(i.usedGrams)>0);
if(!valid.length) return {maxBatches:0,rows:[]};
const maxBatches=Math.floor(Math.min(…valid.map(i=>parseFloat(i.totalGrams)/parseFloat(i.usedGrams))));
const rows=valid.map(i=>{
const total=parseFloat(i.totalGrams),used=parseFloat(i.usedGrams);
const consumed=used*maxBatches, remaining=Math.max(0,total-consumed);
const pricePerG=parseFloat(i.totalPrice)/total;
return { name:i.name||“未命名原料”, total, used, consumed, remaining,
needToBuy:Math.max(0,used-remaining), pricePerG:isNaN(pricePerG)?0:pricePerG, url:i.url, source:i.source };
});
return {maxBatches,rows};
}
